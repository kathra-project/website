# From Ubuntu/Debian Distrib

git clone https://gitlab.com/kathra/deployment.git
cd deployment/terraform/terraform_modules/minikube-stack

# Way 1 - Manual ACME - DNS Challenge

./main.sh deploy --domain=kathra.my-own-domain.org --manual-acme

# Way 2 - Auto ACME with Terraform provider (https://www.terraform.io/docs/providers/acme/dns_providers/)

./main.sh deploy --domain=kathra.my-own-domain.org --acme-dns-provider=ovh --acme-dns-config='{"OVH_APPLICATION_KEY": "app-key", "OVH_APPLICATION_SECRET": "app-secret","OVH_CONSUMER_KEY": "consumer-key","OVH_ENDPOINT": "ovh-eu"}'

# Way 3 - Own Widlcard TLS Certificate (*.kathra.my-own-domain.org)

./main.sh deploy --domain=kathra.my-own-domain.org --tlsCert=file.crt --tlsKey=file.key

# Define env. variables from Azure Account

# Deploy stack into Azure

git clone https://gitlab.com/kathra/deployment.git
cd deployment/terraform/terraform_modules/azure-stack
./main.sh deploy --domain=kathra.my-own-domain.org --azure-group-name=kathra
